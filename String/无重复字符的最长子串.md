class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        if((int)s.size()==0) return 0;
        bool vis[256];
        memset(vis,0,sizeof(vis));
        int L;
        int ans=0;
        L=0;
        for(int i=0;s[i]!='\0';++i)
        {
            // printf("vis[%c]=%d\n",s[i],vis[s[i]]);
            if(vis[s[i]])
            {
                while(vis[s[i]])
                {
                    vis[s[L++]]=false;
                }
            }
            vis[s[i]]=true;
            // printf("L=%d\n",L);
            ans=max(ans,i-L+1);
        }
        return ans;
    }
};
